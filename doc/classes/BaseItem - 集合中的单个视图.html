Parameters:

  • options Object

    [template: 模板字符串]

Example:

   this._initialize({
       template: itemTemp, // 模板字符串
       // 可选
       modelBind: false, // 绑定模型类， 比如文本框内容改变， 模型类相应改变; 当元素为checkbox是， 需设置true-value="01" false-value="00",

       detail: '', // 修改或添加页面地址
       filter: function(model){ // 过滤模型类
       },
       enterRender: '#submit' // 执行回车后的按钮点击的元素选择符
   });
