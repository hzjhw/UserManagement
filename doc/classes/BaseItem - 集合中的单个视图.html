[public] - _dialog

(
  • options
)

Defined in C:\software\WebstormProjects\UserManagement\app\base\BaseItem.js:454

静态对话框， 当你需要显示某个组件的视图但不是以iframe形式打开时

Parameters:

  • options Object

Example:

   this._dialog({
               moduleId: 'SeoDetail', // 模块ID
               title: 'Seo修改', // 对话框标题
               id: this.model.get('id'), // 初始化模块时传入的ID
               width: 600, // 对话框宽度
               height: 250, // 对话框高度
               skin: 'form-horizontal', // className
               button: [ // 自定义按钮
                 {
                   value: '保存',
                   callback: function () {
                   this.title('正在提交..');
                   $("#SeoDetail" + " #submit").click(); // 弹出的对话ID选择符为moduleId值
                   return false; // 去掉此行将直接关闭对话框
                 }}
               ],
               onShow: function(){}, // 对话框弹出后调用
               onClose: function(){
                   ctx._reload(); // 列表刷新
                   ctx.collection.push(Est.cloneDeep(app.getModels())); // 向列表末尾添加数据, 注意必须要深复制
                   ctx.model.set(app.getModels().pop()); // 修改模型类
               }
           });
